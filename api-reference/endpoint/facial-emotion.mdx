Analyze emotions in facial images using our distributed browser network.

## Request

<ParamField body="image" type="blob" required>
  The image file to analyze
</ParamField>

<RequestExample>
```bash
curl -X POST https://api.woolball.xyz/v1/image-facial-emotions \
-H "Authorization: Bearer API_KEY" \
-F "image=@img.png"
```

```javascript
const fs = require('fs');
const FormData = require('form-data');
const fetch = require('node-fetch');

const apiBaseUrl = "https://api.woolball.xyz";
const endpoint = "/v1/image-facial-emotions";

const formData = new FormData();
formData.append('image', fs.createReadStream('img.png'));

fetch(apiBaseUrl + endpoint, {
  method: 'POST',
  headers: {
      'Authorization': 'Bearer API_KEY'
  },
  body: formData
})
.then(res => res.json())
.then(data => {
  console.log("Detected Emotions:", data.data);
})
.catch(console.error);
```

```python
import requests

api_base_url = "https://api.woolball.xyz"
endpoint = "/v1/image-facial-emotions"

headers = {"Authorization": "Bearer API_KEY"}
files = {'image': open('img.png', 'rb')}

response = requests.post(api_base_url + endpoint, headers=headers, files=files)
data = response.json()
print(f"Detected Emotions: {data['data']}")
```
</RequestExample>

## Response

<ResponseField name="success" type="boolean">
  Indicates if the request was successful
</ResponseField>

<ResponseField name="data" type="array">
  Array of detected emotions and their locations
  <Expandable title="properties">
    <ResponseField name="emotion" type="string">
      The detected emotion (e.g., "happy", "sad", "angry", etc.)
    </ResponseField>
    <ResponseField name="confidence" type="number">
      Confidence score between 0 and 1
    </ResponseField>
    <ResponseField name="location" type="object">
      Location of the face in the image
      <Expandable title="properties">
        <ResponseField name="x" type="number">
          X coordinate of the face bounding box
        </ResponseField>
        <ResponseField name="y" type="number">
          Y coordinate of the face bounding box
        </ResponseField>
        <ResponseField name="width" type="number">
          Width of the face bounding box
        </ResponseField>
        <ResponseField name="height" type="number">
          Height of the face bounding box
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>
